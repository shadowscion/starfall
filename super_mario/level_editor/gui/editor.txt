--[[

editor = {}

require("lib/lotus/init.txt")

require("super_mario/level_editor/gui/elements/placement.txt")
require("super_mario/level_editor/gui/elements/selection.txt")
require("super_mario/level_editor/gui/elements/tools.txt")
require("super_mario/level_editor/gui/elements/tile_grid.txt")

--[[

]]
editor.menu = lotus.CreateMenu()

editor.menu:AddContext("background", true)
editor.menu:AddContext("foreground", true)

editor.menu:ClearContext("background", Color(255, 255, 255))


--[[

]]
editor.placement = lotus.CreateElement("placement", editor.menu, "foreground")
editor.selection = lotus.CreateElement("selection", editor.menu, "foreground")
editor.tools = lotus.CreateElement("tools", editor.menu, "foreground")

for _, context in pairs(editor.menu.contexts) do
    for _, element in pairs(context.elements) do
        element:SetSkin(lotus.skins["orange"])
    end
end
